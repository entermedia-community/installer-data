REST_API_Groups
open ${restClient}
echo // Add Group ...
click css=i.request-method-icon.icon-chevron-down
click link=POST
type id=request-url ${currentEnvironment}/media/services/rest/savedata.xml?catalogid=system&searchtype=group&field=name&name.value=TestGroup
click id=request-button
click link=Response Body (Raw)
waitForElementPresent css=#response-body-raw > pre
pause 1000
echo // List Group ...
click css=i.request-method-icon.icon-chevron-down
click link=GET
type id=request-url ${currentEnvironment}/media/services/rest/search.xml?catalogid=system&searchtype=group&operation=exact&field=id&id.value=TestGroup
click id=request-button
click link=Response Body (Raw)
waitForElementPresent css=#response-body-raw > pre
pause 1000
echo // Edit Group ...
click css=i.request-method-icon.icon-chevron-down
click link=POST
type id=request-url ${currentEnvironment}/media/services/rest/savedata.xml?catalogid=system&searchtype=group&id=TestGroup&field=enabled&enabled.value=true
click id=request-button
click link=Response Body (Raw)
waitForElementPresent css=#response-body-raw > pre
pause 1000
echo // List Group Again...
click css=i.request-method-icon.icon-chevron-down
click link=GET
type id=request-url ${currentEnvironment}/media/services/rest/search.xml?catalogid=system&searchtype=group&operation=exact&field=id&id.value=TestGroup
click id=request-button
click link=Response Body (Raw)
waitForElementPresent css=#response-body-raw > pre
pause 1000
echo // Delete Group ...
click css=i.request-method-icon.icon-chevron-down
click link=POST
type id=request-url ${currentEnvironment}/media/services/rest/deletedata.xml?catalogid=system&searchtype=group&id=TestGroup
click id=request-button
click link=Response Body (Raw)
waitForElementPresent css=#response-body-raw > pre
